algoritmo <calcular imposto e custo ao distribuidor de veiculos a partir do custo de fábrica>

	real custoFabrica, impostos, distribuidor, total

    escreva("Qual o custo de fábrica: ")

    distribuidor = (custoFabrica + impostos) * 0.28
